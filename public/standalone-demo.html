This demo shows how to integrate MorphChat into any HTML page without React or build tools.

Integration Code:

<!-- Load MorphChat Widget -->
<script src="https://your-app.herokuapp.com/widget.js"></script>

<script>
    MorphChat.init({
        apiKey: 'your-openai-api-key',
        theme: 'purple',
        botProfile: {
            name: 'MorphChat Assistant'
        },
        prompt: {
            welcomeMessage: 'Hello! How can I help you today?',
            model: 'gpt-4o-mini'
        }
    });
</script>

ðŸ”§ Configuration Options

MorphChat.init({
    // Widget positioning
    corner: "right", // "left" | "right"
    
    // Theme configuration
    theme: "purple", // "purple" | "blue" | "green" | "red" | "orange" | "pink" | "gray"
    mode: "auto", // "light" | "dark" | "auto"
    
    // Bot profile
    botProfile: {
        name: "MorphChat",
        avatar: "https://example.com/avatar.png", // Optional
        showAvatar: true
    },
    
    // AI configuration
    prompt: {
        apiKey: "your-openai-api-key",
        welcomeMessage: "Hello! I'm MorphChat, your AI assistant.",
        instructions: "You are a helpful AI assistant.",
        model: "gpt-4o-mini",
        timeout: 30000,
        localStorage: true
    },
    
    // Status
    status: {
        isOnline: true,
        maintenanceMode: false
    }
});
